CREATE DATABASE exercises;
USE exercises;

-- UNSIGNED: prevent negative numbers
CREATE TABLE student (
	id int auto_increment,
	full_name VARCHAR(255) NOT NULL,
	gender VARCHAR(10),
	age int UNSIGNED,
	city VARCHAR(50),
	weight FLOAT(5, 2) UNSIGNED,
	
	primary key(id)
);
